## 1. 背景与目标

### 1.1 为什么要拆分

MMOItems 是一个功能极其丰富但过于臃肿的插件，存在以下问题：

| 问题 | 说明 |
|------|------|
| **代码耦合** | 各功能模块深度耦合，难以单独维护 |
| **调试困难** | 一个bug可能影响整个插件 |
| **扩展不灵活** | 用户无法按需启用功能，必须加载全部 |
| **学习曲线陡峭** | 新人难以理解整体架构 |
| **性能开销** | 不使用的功能也在占用资源 |

### 1.2 拆分目标

```
┌─────────────────────────────────────────────────────────┐
│                    单体插件 (MMOItems)                   │
│  物品系统 + 锻造 + 宝石 + 重铸 + 升级 + 方块 + ...       │
└─────────────────────────────────────────────────────────┘
                           │
                           ▼
┌─────────────────────────────────────────────────────────┐
│                    Core-API (核心API)                    │
│  所有插件的公共基础                                       │
└─────────────────────────────────────────────────────────┘
         │         │         │         │         │
         ▼         ▼         ▼         ▼         ▼
    ┌─────────┐ ┌─────────┐ ┌─────────┐ ┌─────────┐ ┌─────────┐
    │ Items   │ │ Forge   │ │ Gems    │ │ Reforge │ │ Upgrade │
    │ 物品系统 │ │ 锻造系统 │ │ 宝石系统 │ │ 重铸系统 │ │ 升级系统 │
    └─────────┘ └─────────┘ └─────────┘ └─────────┘ └─────────┘
         │         │         │         │         │
         ▼         ▼         ▼         ▼         ▼
    ┌─────────┐ ┌─────────┐ ┌─────────┐ ┌─────────┐ ┌─────────┐
    │ Blocks  │ │ Sets    │ │ Skills  │ │ Drops   │ │ ...     │
    │ 自定义方块│ │ 物品套装 │ │ 技能系统 │ │ 掉落表  │ │ 其他扩展 │
    └─────────┘ └─────────┘ └─────────┘ └─────────┘ └─────────┘
```

### 1.3 设计原则

1. **单一职责**：每个插件只负责一个核心功能
2. **松耦合**：插件间通过API和事件通信，不直接依赖实现
3. **可独立运行**：除了Core-API，各插件应能独立工作
4. **向后兼容**：配置文件格式尽量保持与MMOItems一致
5. **按需加载**：用户可以选择只安装需要的插件

### 6.2 数据模型

#### 6.2.1 CraftingStatus - 玩家锻造状态

```java
/**
 * 玩家锻造状态
 * 每个玩家一个实例，管理该玩家所有锻造台的队列
 */
public class CraftingStatus {
    
    // 按锻造台ID分类存储队列
    private final Map<String, CraftingQueue> queues = new HashMap<>();
    
    /**
     * 获取或创建指定锻造台的队列
     */
    public CraftingQueue getQueue(String stationId) {
        return queues.computeIfAbsent(stationId, CraftingQueue::new);
    }
    
    /**
     * 保存到配置
     */
    public void save(ConfigurationSection section) {
        for (Map.Entry<String, CraftingQueue> entry : queues.entrySet()) {
            String stationId = entry.getKey();
            CraftingQueue queue = entry.getValue();
            
            for (QueueItem item : queue.getItems()) {
                String path = stationId + "." + item.getUuid().toString();
                section.set(path + ".recipe", item.getRecipeId());
                section.set(path + ".start", item.getStart());
                section.set(path + ".completion", item.getCompletion());
            }
        }
    }
    
    /**
     * 从配置加载
     */
    public void load(ConfigurationSection section) {
        for (String stationId : section.getKeys(false)) {
            CraftingQueue queue = new CraftingQueue(stationId);
            queues.put(stationId, queue);
            
            ConfigurationSection stationSection = section.getConfigurationSection(stationId);
            for (String itemKey : stationSection.getKeys(false)) {
                String recipeId = stationSection.getString(itemKey + ".recipe");
                long start = stationSection.getLong(itemKey + ".start");
                long completion = stationSection.getLong(itemKey + ".completion");
                queue.add(new QueueItem(recipeId, UUID.fromString(itemKey), start, completion));
            }
        }
    }
}
```

#### 6.2.2 CraftingQueue - 锻造队列

```java
/**
 * 单个锻造台的锻造队列
 */
public class CraftingQueue {
    
    private final String stationId;
    private final List<QueueItem> items = new ArrayList<>();
    
    public CraftingQueue(String stationId) {
        this.stationId = stationId;
    }
    
    /**
     * 检查队列是否已满
     */
    public boolean isFull(int maxSize) {
        return items.size() >= maxSize;
    }
    
    /**
     * 添加新配方到队列
     * 关键逻辑：计算完成时间
     */
    public QueueItem add(Recipe recipe) {
        long craftingTimeMs = (long) (recipe.getCraftingTime() * 1000);
        long start = System.currentTimeMillis();
        
        // 如果队列不为空，完成时间 = 上一个完成时间 + 当前配方时间
        long completion = items.isEmpty() 
            ? start + craftingTimeMs 
            : items.get(items.size() - 1).getCompletion() + craftingTimeMs;
        
        QueueItem item = new QueueItem(
            recipe.getId(), 
            UUID.randomUUID(), 
            start, 
            completion
        );
        items.add(item);
        return item;
    }
    
    /**
     * 移除队列项（领取或取消）
     * 关键逻辑：后续项时间提前
     */
    public void remove(UUID itemId, long craftingTimeMs) {
        int index = findIndex(itemId);
        if (index < 0) return;
        
        QueueItem removed = items.remove(index);
        
        // 计算节省的时间 = min(剩余时间, 配方时间)
        long savedTime = Math.min(removed.getRemaining(), craftingTimeMs);
        
        // 后续所有项的完成时间提前
        for (int i = index; i < items.size(); i++) {
            items.get(i).advanceCompletion(savedTime);
        }
    }
    
    /**
     * 获取指定UUID的队列项
     */
    public Optional<QueueItem> getItem(UUID uuid) {
        return items.stream()
            .filter(item -> item.getUuid().equals(uuid))
            .findFirst();
    }
    
    private int findIndex(UUID itemId) {
        for (int i = 0; i < items.size(); i++) {
            if (items.get(i).getUuid().equals(itemId)) {
                return i;
            }
        }
        return -1;
    }
    
    // Getters
    public List<QueueItem> getItems() { return Collections.unmodifiableList(items); }
    public int size() { return items.size(); }
    public String getStationId() { return stationId; }
}
```

#### 6.2.3 QueueItem - 队列项

```java
/**
 * 锻造队列中的单个项
 * 使用时间戳计算状态，无需定时任务
 */
public class QueueItem {
    
    private final String recipeId;
    private final UUID uuid;
    private final long start;
    private long completion;  // 完成时间戳（可变）
    
    public QueueItem(String recipeId, UUID uuid, long start, long completion) {
        this.recipeId = recipeId;
        this.uuid = uuid;
        this.start = start;
        this.completion = completion;
    }
    
    /**
     * 是否已完成
     */
    public boolean isReady() {
        return getRemaining() <= 0;
    }
    
    /**
     * 获取剩余时间（毫秒）
     */
    public long getRemaining() {
        return Math.max(0, completion - System.currentTimeMillis());
    }
    
    /**
     * 获取已用时间（毫秒）
     */
    public long getElapsed() {
        return System.currentTimeMillis() - start;
    }
    
    /**
     * 获取完成进度（0.0 - 1.0）
     */
    public double getProgress(long totalTimeMs) {
        if (totalTimeMs <= 0) return 1.0;
        double progress = (double) getElapsed() / totalTimeMs;
        return Math.min(1.0, Math.max(0.0, progress));
    }
    
    /**
     * 提前完成时间（当前面的项被取消时）
     */
    public void advanceCompletion(long amount) {
        this.completion -= amount;
    }
    
    // Getters
    public String getRecipeId() { return recipeId; }
    public UUID getUuid() { return uuid; }
    public long getStart() { return start; }
    public long getCompletion() { return completion; }
}
```

### 6.3 时间计算逻辑

```
场景1: 空队列添加配方A（耗时10秒）
时间线:
当前:  1000ms
start: 1000ms
completion: 1000 + 10000 = 11000ms

场景2: 在A完成前添加配方B（耗时5秒）
时间线:
当前:  3000ms (A已用2秒，剩余8秒)
B的start: 3000ms
B的completion: A.completion + 5000 = 11000 + 5000 = 16000ms

场景3: 取消配方A（此时A还剩8秒）
时间线:
移除A后，节省的时间 = min(8000, 10000) = 8000ms
B的completion = 16000 - 8000 = 8000ms
```

### 6.4 锻造流程

```

         │ 不满足
         ├─────────► 返回错误消息
         │ 满足
```

### 6.8 GUI显示刷新

```java
/**
 * 队列显示刷新任务
 * 只在玩家打开GUI时运行
 */
public class QueueDisplayRefresher extends BukkitRunnable {
    
    private final Inventory inventory;
    private final CraftingQueue queue;
    private final Layout layout;
    private final ItemRegistry itemRegistry;
    private int offset = 0;
    
    @Override
    public void run() {
        // 检查是否还有查看者
        if (inventory.getViewers().isEmpty()) {
            cancel();
            return;
        }
        
        // 更新队列物品显示
        List<Integer> queueSlots = layout.getQueueSlots();
        for (int i = 0; i < queueSlots.size(); i++) {
            int queueIndex = offset + i;
            int slot = queueSlots.get(i);
            
            if (queueIndex >= queue.getItems().size()) {
                inventory.setItem(slot, layout.getNoQueueItem());
            } else {
                QueueItem item = queue.getItems().get(queueIndex);
                inventory.setItem(slot, createQueueDisplay(item, queueIndex + 1));
            }
        }
    }
    
    /**
     * 创建队列显示物品
     */
    private ItemStack createQueueDisplay(QueueItem item, int position) {
        // 获取配方预览
        Optional<Recipe> recipeOpt = getRecipe(item.getRecipeId());
        if (!recipeOpt.isPresent()) {
            return new ItemStack(Material.BARRIER);
        }
        
        Recipe recipe = recipeOpt.get();
        ItemStack display = recipe.getOutput(null);
        
        // 设置名称和Lore
        ItemMeta meta = display.getItemMeta();
        List<String> lore = meta.hasLore() ? new ArrayList<>(meta.getLore()) : new ArrayList<>();
        
        lore.add("");
        if (item.isReady()) {
            lore.add(ChatColor.GREEN + "已完成！点击领取");
        } else {
            lore.add(ChatColor.YELLOW + "剩余时间: " + formatTime(item.getRemaining()));
            lore.add(ChatColor.GRAY + "点击取消");
        }
        
        meta.setLore(lore);
        display.setItemMeta(meta);
        
        // 写入NBT标识
        return NBTItem.get(display)
            .addTag(new ItemTag("queueId", item.getUuid().toString()))
            .toItem();
    }
}
```

### 6.9 时间格式化工具

```java
public class TimeFormatter {
    
    private static final long[] UNITS = {
        24 * 60 * 60 * 1000,  // 天
        60 * 60 * 1000,       // 时
        60 * 1000,            // 分
        1000                  // 秒
    };
    
    private static final String[] SUFFIXES = {"d", "h", "m", "s"};
    
    /**
     * 格式化剩余时间（最多显示两个单位）
     */
    public static String format(long ms) {
        if (ms <= 0) return "0s";
        
        StringBuilder result = new StringBuilder();
        int count = 0;
        
        for (int i = 0; i < UNITS.length && count < 2; i++) {
            if (ms >= UNITS[i]) {
                long value = ms / UNITS[i];
                result.append(value).append(SUFFIXES[i]).append(" ");
                ms %= UNITS[i];
                count++;
            }
        }
        
        return result.length() > 0 ? result.toString().trim() : "1s";
    }
    
    // 示例:
    // format(3661000) -> "1h 1m"
    // format(150000)  -> "2m 30s"
    // format(5000)    -> "5s"
}
```
